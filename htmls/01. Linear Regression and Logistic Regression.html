# train the model
losses = []
for i in range(1000):
    y_pred = model(x)
    loss = criterion(y_pred, y)
    losses.append(loss.item())
    
    optimizer.zero_grad()
    loss.backward()
    optimizer.step()
    
# make prediction
with torch.no_grad():
    y_pred = model(x).view(-1).numpy()
